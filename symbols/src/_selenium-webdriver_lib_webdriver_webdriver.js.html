 47 /**
 48  * Creates a new WebDriver client, which provides control over a browser.
 49  *
 50  * Every WebDriver command returns a {@code webdriver.promise.Promise} that
 51  * represents the result of that command. Callbacks may be registered on this
 52  * object to manipulate the command result or catch an expected error. Any
 53  * commands scheduled with a callback are considered sub-commands and will
 54  * execute before the next command in the current frame. For example:
 55  * <pre><code>
 56  *   var message = [];
 57  *   driver.call(message.push, message, 'a').then(function() {
 58  *     driver.call(message.push, message, 'b');
 59  *   });
 60  *   driver.call(message.push, message, 'c');
 61  *   driver.call(function() {
 62  *     alert('message is abc? ' + (message.join('') == 'abc'));
 63  *   });
 64  * </code></pre>
 65  *
 66  * @param {!(webdriver.Session|webdriver.promise.Promise)} session Either a
 67  *     known session or a promise that will be resolved to a session.
 68  * @param {!webdriver.CommandExecutor} executor The executor to use when
 69  *     sending commands to the browser.
 70  * @param {webdriver.promise.ControlFlow=} opt_flow The flow to
 71  *     schedule commands through. Defaults to the active flow object.
 72  * @constructor
 73  */
 74 webdriver.WebDriver = function(session, executor, opt_flow) {
 75 
 76   /** @private {!(webdriver.Session|webdriver.promise.Promise)} */
 77   this.session_ = session;
 78 
 79   /** @private {!webdriver.CommandExecutor} */
 80   this.executor_ = executor;
 81 
 82   /** @private {!webdriver.promise.ControlFlow} */
 83   this.flow_ = opt_flow || webdriver.promise.controlFlow();
 84 };

118 /**
119  * Sends a command to the server that is expected to return the details for a
120  * {@link webdriver.Session}. This may either be an existing session, or a
121  * newly created one.
122  * @param {!webdriver.CommandExecutor} executor Command executor to use when
123  *     querying for session details.
124  * @param {!webdriver.Command} command The command to send to fetch the session
125  *     details.
126  * @param {string} description A descriptive debug label for this action.
127  * @return {!webdriver.WebDriver} A new WebDriver client for the session.
128  * @private
129  */
130 webdriver.WebDriver.acquireSession_ = function(executor, command, description) {
131   var session = webdriver.promise.controlFlow().execute(function() {
132     return webdriver.WebDriver.executeCommand_(executor, command).
133         then(function(response) {
134           bot.response.checkResponse(response);
135           return new webdriver.Session(response['sessionId'],
136               response['value']);
137         });
138   }, description);
139   return new webdriver.WebDriver(session, executor);
140 };

143 /**
144  * Converts an object to its JSON representation in the WebDriver wire protocol.
145  * When converting values of type object, the following steps will be taken:
146  * <ol>
147  * <li>if the object provides a "toWireValue" function, the return value will
148  *     be returned in its fully resolved state (e.g. this function may return
149  *     promise values)</li>
150  * <li>if the object provides a "toJSON" function, the return value of this
151  *     function will be returned</li>
152  * <li>otherwise, the value of each key will be recursively converted according
153  *     to the rules above.</li>
154  * </ol>
155  *
156  * @param {*} obj The object to convert.
157  * @return {!webdriver.promise.Promise} A promise that will resolve to the
158  *     input value's JSON representation.
159  * @private
160  * @see http://code.google.com/p/selenium/wiki/JsonWireProtocol
161  */
162 webdriver.WebDriver.toWireValue_ = function(obj) {
163   switch (goog.typeOf(obj)) {
164     case 'array':
165       return webdriver.promise.fullyResolved(
166           goog.array.map(/** @type {!Array} */ (obj),
167               webdriver.WebDriver.toWireValue_));
168     case 'object':
169       if (goog.isFunction(obj.toWireValue)) {
170         return webdriver.promise.fullyResolved(obj.toWireValue());
171       }
172       if (goog.isFunction(obj.toJSON)) {
173         return webdriver.promise.fulfilled(obj.toJSON());
174       }
175       if (goog.isNumber(obj.nodeType) && goog.isString(obj.nodeName)) {
176         throw Error([
177           'Invalid argument type: ', obj.nodeName, '(', obj.nodeType, ')'
178         ].join(''));
179       }
180       return webdriver.promise.fullyResolved(
181           goog.object.map(/** @type {!Object} */ (obj),
182               webdriver.WebDriver.toWireValue_));
183     case 'function':
184       return webdriver.promise.fulfilled('' + obj);
185     case 'undefined':
186       return webdriver.promise.fulfilled(null);
187     default:
188       return webdriver.promise.fulfilled(obj);
189   }
190 };

193 /**
194  * Converts a value from its JSON representation according to the WebDriver wire
195  * protocol. Any JSON object containing a
196  * {@code webdriver.WebElement.ELEMENT_KEY} key will be decoded to a
197  * {@code webdriver.WebElement} object. All other values will be passed through
198  * as is.
199  * @param {!webdriver.WebDriver} driver The driver instance to use as the
200  *     parent of any unwrapped {@code webdriver.WebElement} values.
201  * @param {*} value The value to convert.
202  * @return {*} The converted value.
203  * @see http://code.google.com/p/selenium/wiki/JsonWireProtocol
204  * @private
205  */
206 webdriver.WebDriver.fromWireValue_ = function(driver, value) {
207   if (goog.isArray(value)) {
208     value = goog.array.map(/**@type {goog.array.ArrayLike}*/ (value),
209         goog.partial(webdriver.WebDriver.fromWireValue_, driver));
210   } else if (value && goog.isObject(value) && !goog.isFunction(value)) {
211     if (webdriver.WebElement.ELEMENT_KEY in value) {
212       value = new webdriver.WebElement(driver,
213           value[webdriver.WebElement.ELEMENT_KEY]);
214     } else {
215       value = goog.object.map(/**@type {!Object}*/ (value),
216           goog.partial(webdriver.WebDriver.fromWireValue_, driver));
217     }
218   }
219   return value;
220 };

223 /**
224  * Translates a command to its wire-protocol representation before passing it
225  * to the given {@code executor} for execution.
226  * @param {!webdriver.CommandExecutor} executor The executor to use.
227  * @param {!webdriver.Command} command The command to execute.
228  * @return {!webdriver.promise.Promise} A promise that will resolve with the
229  *     command response.
230  * @private
231  */
232 webdriver.WebDriver.executeCommand_ = function(executor, command) {
233   return webdriver.promise.fullyResolved(command.getParameters()).
234       then(webdriver.WebDriver.toWireValue_).
235       then(function(parameters) {
236         command.setParameters(parameters);
237         return webdriver.promise.checkedNodeCall(
238             goog.bind(executor.execute, executor, command));
239       });
240 };

252 /**
253  * Schedules a {@code webdriver.Command} to be executed by this driver's
254  * {@code webdriver.CommandExecutor}.
255  * @param {!webdriver.Command} command The command to schedule.
256  * @param {string} description A description of the command for debugging.
257  * @return {!webdriver.promise.Promise} A promise that will be resolved with
258  *     the command result.
259  */
260 webdriver.WebDriver.prototype.schedule = function(command, description) {
261   var self = this;
262 
263   checkHasNotQuit();
264   command.setParameter('sessionId', this.session_);
265 
266   var flow = this.flow_;
267   return flow.execute(function() {
268     // A call to WebDriver.quit() may have been scheduled in the same event
269     // loop as this |command|, which would prevent us from detecting that the
270     // driver has quit above.  Therefore, we need to make another quick check.
271     // We still check above so we can fail as early as possible.
272     checkHasNotQuit();
273     return webdriver.WebDriver.executeCommand_(self.executor_, command);
274   }, description).then(function(response) {
275     try {
276       bot.response.checkResponse(response);
277     } catch (ex) {
278       var value = response['value'];
279       if (ex.code === bot.ErrorCode.MODAL_DIALOG_OPENED) {
280         var text = value && value['alert'] ? value['alert']['text'] : '';
281         throw new webdriver.UnhandledAlertError(ex.message,
282             new webdriver.Alert(self, text));
283       }
284       throw ex;
285     }
286     return webdriver.WebDriver.fromWireValue_(self, response['value']);
287   });
288 
289   function checkHasNotQuit() {
290     if (!self.session_) {
291       throw new Error('This driver instance does not have a valid session ID ' +
292                       '(did you call WebDriver.quit()?) and may no longer be ' +
293                       'used.');
294     }
295   }
296 };

346   var result = this.schedule(

375 /**
376  * Schedules a command to execute JavaScript in the context of the currently
377  * selected frame or window. The script fragment will be executed as the body
378  * of an anonymous function. If the script is provided as a function object,
379  * that function will be converted to a string for injection into the target
380  * window.
381  *
382  * Any arguments provided in addition to the script will be included as script
383  * arguments and may be referenced using the {@code arguments} object.
384  * Arguments may be a boolean, number, string, or {@code webdriver.WebElement}.
385  * Arrays and objects may also be used as script arguments as long as each item
386  * adheres to the types previously mentioned.
387  *
388  * The script may refer to any variables accessible from the current window.
389  * Furthermore, the script will execute in the window's context, thus
390  * {@code document} may be used to refer to the current document. Any local
391  * variables will not be available once the script has finished executing,
392  * though global variables will persist.
393  *
394  * If the script has a return value (i.e. if the script contains a return
395  * statement), then the following steps will be taken for resolving this
396  * functions return value:
397  * <ul>
398  * <li>For a HTML element, the value will resolve to a
399  *     {@code webdriver.WebElement}</li>
400  * <li>Null and undefined return values will resolve to null</li>
401  * <li>Booleans, numbers, and strings will resolve as is</li>
402  * <li>Functions will resolve to their string representation</li>
403  * <li>For arrays and objects, each member item will be converted according to
404  *     the rules above</li>
405  * </ul>
406  *
407  * @param {!(string|Function)} script The script to execute.
408  * @param {...*} var_args The arguments to pass to the script.
409  * @return {!webdriver.promise.Promise} A promise that will resolve to the
410  *    scripts return value.
411  */
412 webdriver.WebDriver.prototype.executeScript = function(script, var_args) {
413   if (goog.isFunction(script)) {
414     script = 'return (' + script + ').apply(null, arguments);';
415   }
416   return this.schedule(
417       new webdriver.Command(webdriver.CommandName.EXECUTE_SCRIPT).
418           setParameter('script', script).
419           setParameter('args', goog.array.slice(arguments, 1)),
420       'WebDriver.executeScript()');
421 };

424 /**
425  * Schedules a command to execute asynchronous JavaScript in the context of the
426  * currently selected frame or window. The script fragment will be executed as
427  * the body of an anonymous function. If the script is provided as a function
428  * object, that function will be converted to a string for injection into the
429  * target window.
430  *
431  * Any arguments provided in addition to the script will be included as script
432  * arguments and may be referenced using the {@code arguments} object.
433  * Arguments may be a boolean, number, string, or {@code webdriver.WebElement}.
434  * Arrays and objects may also be used as script arguments as long as each item
435  * adheres to the types previously mentioned.
436  *
437  * Unlike executing synchronous JavaScript with
438  * {@code webdriver.WebDriver.prototype.executeScript}, scripts executed with
439  * this function must explicitly signal they are finished by invoking the
440  * provided callback. This callback will always be injected into the
441  * executed function as the last argument, and thus may be referenced with
442  * {@code arguments[arguments.length - 1]}. The following steps will be taken
443  * for resolving this functions return value against the first argument to the
444  * script's callback function:
445  * <ul>
446  * <li>For a HTML element, the value will resolve to a
447  *     {@code webdriver.WebElement}</li>
448  * <li>Null and undefined return values will resolve to null</li>
449  * <li>Booleans, numbers, and strings will resolve as is</li>
450  * <li>Functions will resolve to their string representation</li>
451  * <li>For arrays and objects, each member item will be converted according to
452  *     the rules above</li>
453  * </ul>
454  *
455  * Example #1: Performing a sleep that is synchronized with the currently
456  * selected window:
457  * <code><pre>
458  * var start = new Date().getTime();
459  * driver.executeAsyncScript(
460  *     'window.setTimeout(arguments[arguments.length - 1], 500);').
461  *     then(function() {
462  *       console.log('Elapsed time: ' + (new Date().getTime() - start) + ' ms');
463  *     });
464  * </pre></code>
465  *
466  * Example #2: Synchronizing a test with an AJAX application:
467  * <code><pre>
468  * var button = driver.findElement(By.id('compose-button'));
469  * button.click();
470  * driver.executeAsyncScript(
471  *     'var callback = arguments[arguments.length - 1];' +
472  *     'mailClient.getComposeWindowWidget().onload(callback);');
473  * driver.switchTo().frame('composeWidget');
474  * driver.findElement(By.id('to')).sendKEys('dog@example.com');
475  * </pre></code>
476  *
477  * Example #3: Injecting a XMLHttpRequest and waiting for the result. In this
478  * example, the inject script is specified with a function literal. When using
479  * this format, the function is converted to a string for injection, so it
480  * should not reference any symbols not defined in the scope of the page under
481  * test.
482  * <code><pre>
483  * driver.executeAsyncScript(function() {
484  *   var callback = arguments[arguments.length - 1];
485  *   var xhr = new XMLHttpRequest();
486  *   xhr.open("GET", "/resource/data.json", true);
487  *   xhr.onreadystatechange = function() {
488  *     if (xhr.readyState == 4) {
489  *       callback(xhr.resposneText);
490  *     }
491  *   }
492  *   xhr.send('');
493  * }).then(function(str) {
494  *   console.log(JSON.parse(str)['food']);
495  * });
496  * </pre></code>
497  *
498  * @param {!(string|Function)} script The script to execute.
499  * @param {...*} var_args The arguments to pass to the script.
500  * @return {!webdriver.promise.Promise} A promise that will resolve to the
501  *    scripts return value.
502  */
503 webdriver.WebDriver.prototype.executeAsyncScript = function(script, var_args) {
504   if (goog.isFunction(script)) {
505     script = 'return (' + script + ').apply(null, arguments);';
506   }
507   return this.schedule(
508       new webdriver.Command(webdriver.CommandName.EXECUTE_ASYNC_SCRIPT).
509           setParameter('script', script).
510           setParameter('args', goog.array.slice(arguments, 1)),
511       'WebDriver.executeScript()');
512 };

645 /**
646  * Schedule a command to find an element on the page. If the element cannot be
647  * found, a {@code bot.ErrorCode.NO_SUCH_ELEMENT} result will be returned
648  * by the driver. Unlike other commands, this error cannot be suppressed. In
649  * other words, scheduling a command to find an element doubles as an assert
650  * that the element is present on the page. To test whether an element is
651  * present on the page, use {@code #isElementPresent} instead.
652  *
653  * <p>The search criteria for find an element may either be a
654  * {@code webdriver.Locator} object, or a simple JSON object whose sole key
655  * is one of the accepted locator strategies, as defined by
656  * {@code webdriver.Locator.Strategy}. For example, the following two statements
657  * are equivalent:
658  * <code><pre>
659  * var e1 = driver.findElement(By.id('foo'));
660  * var e2 = driver.findElement({id:'foo'});
661  * </pre></code>
662  *
663  * <p>When running in the browser, a WebDriver cannot manipulate DOM elements
664  * directly; it may do so only through a {@link webdriver.WebElement} reference.
665  * This function may be used to generate a WebElement from a DOM element. A
666  * reference to the DOM element will be stored in a known location and this
667  * driver will attempt to retrieve it through {@link #executeScript}. If the
668  * element cannot be found (eg, it belongs to a different document than the
669  * one this instance is currently focused on), a
670  * {@link bot.ErrorCode.NO_SUCH_ELEMENT} error will be returned.
671  *
672  * @param {!(webdriver.Locator|Object.<string>|Element)} locatorOrElement The
673  *     locator strategy to use when searching for the element, or the actual
674  *     DOM element to be located by the server.
675  * @param {...} var_args Arguments to pass to {@code #executeScript} if using a
676  *     JavaScript locator.  Otherwise ignored.
677  * @return {!webdriver.WebElement} A WebElement that can be used to issue
678  *     commands against the located element. If the element is not found, the
679  *     element will be invalidated and all scheduled commands aborted.
680  */
681 webdriver.WebDriver.prototype.findElement = function(locatorOrElement,
682                                                      var_args) {
683   var id;
684   if (locatorOrElement.nodeType === 1 && locatorOrElement.ownerDocument) {
685     var element = /** @type {!Element} */ (locatorOrElement);
686     id = this.findDomElement_(element).
687         then(function(elements) {
688           if (!elements.length) {
689             throw new bot.Error(bot.ErrorCode.NO_SUCH_ELEMENT,
690                 'Unable to locate element. Is WebDriver focused on its ' +
691                     'ownerDocument\'s frame?');
692           }
693           return elements[0];
694         });
695   } else {
696     var locator = webdriver.Locator.checkLocator(locatorOrElement);
697     if (locator.using == 'js') {
698       var args = goog.array.slice(arguments, 1);
699       goog.array.splice(args, 0, 0, locator.value);
700 
701       id = this.executeScript.apply(this, args).
702           then(function(value) {
703             if (!(value instanceof webdriver.WebElement)) {
704               throw new Error('JS locator script result was not a WebElement');
705             }
706             return value;
707           });
708     } else {
709       var command = new webdriver.Command(webdriver.CommandName.FIND_ELEMENT).
710           setParameter('using', locator.using).
711           setParameter('value', locator.value);
712       id = this.schedule(command, 'WebDriver.findElement(' + locator + ')');
713     }
714   }
715   return new webdriver.WebElement(this, id);
716 };

719 /**
720  * Locates a DOM element so that commands may be issued against it using the
721  * {@link webdriver.WebElement} class. This is accomplished by storing a
722  * reference to the element in an object on the element's ownerDocument.
723  * {@link #executeScript} will then be used to create a WebElement from this
724  * reference. This requires this driver to currently be focused on the
725  * ownerDocument's window+frame.
726 
727  * @param {!Element} element The element to locate.
728  * @return {!webdriver.promise.Promise} A promise that will be resolved
729  *     with the located WebElement.
730  * @private
731  */
732 webdriver.WebDriver.prototype.findDomElement_ = function(element) {
733   var doc = element.ownerDocument;
734   var store = doc['$webdriver$'] = doc['$webdriver$'] || {};
735   var id = Math.floor(Math.random() * goog.now()).toString(36);
736   store[id] = element;
737   element[id] = id;
738 
739   function cleanUp() {
740     delete store[id];
741   }
742 
743   function lookupElement(id) {
744     var store = document['$webdriver$'];
745     if (!store) {
746       return null;
747     }
748 
749     var element = store[id];
750     if (!element || element[id] !== id) {
751       return [];
752     }
753     return [element];
754   }
755 
756   return this.executeScript(lookupElement, id).
757       then(function(value) {
758         cleanUp();
759         if (value.length && !(value[0] instanceof webdriver.WebElement)) {
760           throw new Error('JS locator script result was not a WebElement');
761         }
762         return value;
763       }, cleanUp);
764 };

768  * Schedules a command to test if an element is present on the page.
769  *
770  * <p>If given a DOM element, this function will check if it belongs to the
771  * document the driver is currently focused on. Otherwise, the function will
772  * test if at least one element can be found with the given search criteria.
773  *
774  * @param {!(webdriver.Locator|Object.<string>|Element)} locatorOrElement The
775  *     locator strategy to use when searching for the element, or the actual
776  *     DOM element to be located by the server.
777  * @param {...} var_args Arguments to pass to {@code #executeScript} if using a
778  *     JavaScript locator.  Otherwise ignored.

799  * @param {...} var_args Arguments to pass to {@code #executeScript} if using a
800  *     JavaScript locator.  Otherwise ignored.
801  * @return {!webdriver.promise.Promise} A promise that will be resolved to an

823         setParameter('using', locator.using).

953  * Schedules a command to add a cookie.
954  * @param {string} name The cookie name.
955  * @param {string} value The cookie value.
956  * @param {string=} opt_path The cookie path.
957  * @param {string=} opt_domain The cookie domain.
958  * @param {boolean=} opt_isSecure Whether the cookie is secure.
959  * @param {(number|!Date)=} opt_expiry When the cookie expires. If specified as
960  *     a number, should be in milliseconds since midnight, January 1, 1970 UTC.
961  * @return {!webdriver.promise.Promise} A promise that will be resolved when the
962  *     cookie has been added to the page.
963  */
964 webdriver.WebDriver.Options.prototype.addCookie = function(
965     name, value, opt_path, opt_domain, opt_isSecure, opt_expiry) {
966   // We do not allow '=' or ';' in the name.
967   if (/[;=]/.test(name)) {
968     throw Error('Invalid cookie name "' + name + '"');
969   }
970 
971   // We do not allow ';' in value.
972   if (/;/.test(value)) {
973     throw Error('Invalid cookie value "' + value + '"');
974   }
975 
976   var cookieString = name + '=' + value +
977       (opt_domain ? ';domain=' + opt_domain : '') +
978       (opt_path ? ';path=' + opt_path : '') +
979       (opt_isSecure ? ';secure' : '');
980 
981   var expiry;
982   if (goog.isDef(opt_expiry)) {
983     var expiryDate;
984     if (goog.isNumber(opt_expiry)) {
985       expiryDate = new Date(opt_expiry);
986     } else {
987       expiryDate = /** @type {!Date} */ (opt_expiry);
988       opt_expiry = expiryDate.getTime();
989     }
990     cookieString += ';expires=' + expiryDate.toUTCString();
991     // Convert from milliseconds to seconds.
992     expiry = Math.floor(/** @type {number} */ (opt_expiry) / 1000);
993   }
994 
995   return this.driver_.schedule(
996       new webdriver.Command(webdriver.CommandName.ADD_COOKIE).
997           setParameter('cookie', {
998             'name': name,
999             'value': value,
1000             'path': opt_path,
1001             'domain': opt_domain,
1002             'secure': !!opt_isSecure,
1003             'expiry': expiry
1004           }),
1005       'WebDriver.manage().addCookie(' + cookieString + ')');
1006 };

1450 /**
1451  * Represents a DOM element. WebElements can be found by searching from the
1452  * document root using a {@code webdriver.WebDriver} instance, or by searching
1453  * under another {@code webdriver.WebElement}:
1454  * <pre><code>
1455  *   driver.get('http://www.google.com');
1456  *   var searchForm = driver.findElement(By.tagName('form'));
1457  *   var searchBox = searchForm.findElement(By.name('q'));
1458  *   searchBox.sendKeys('webdriver');
1459  * </code></pre>
1460  *
1461  * The WebElement is implemented as a promise for compatibility with the promise
1462  * API. It will always resolve itself when its internal state has been fully
1463  * resolved and commands may be issued against the element. This can be used to
1464  * catch errors when an element cannot be located on the page:
1465  * <pre><code>
1466  *   driver.findElement(By.id('not-there')).then(function(element) {
1467  *     alert('Found an element that was not expected to be there!');
1468  *   }, function(error) {
1469  *     alert('The element was not found, as expected');
1470  *   });
1471  * </code></pre>
1472  *
1473  * @param {!webdriver.WebDriver} driver The parent WebDriver instance for this
1474  *     element.
1475  * @param {!(string|webdriver.promise.Promise)} id Either the opaque ID for the
1476  *     underlying DOM element assigned by the server, or a promise that will
1477  *     resolve to that ID or another WebElement.
1478  * @constructor
1479  * @extends {webdriver.promise.Deferred}
1480  */
1481 webdriver.WebElement = function(driver, id) {
1482   webdriver.promise.Deferred.call(this, null, driver.controlFlow());
1483 
1484   /**
1485    * The parent WebDriver instance for this element.
1486    * @private {!webdriver.WebDriver}
1487    */
1488   this.driver_ = driver;
1489 
1490   // This class is responsible for resolving itself; delete the resolve and
1491   // reject methods so they may not be accessed by consumers of this class.
1492   var fulfill = goog.partial(this.fulfill, this);
1493   var reject = this.reject;
1494   delete this.promise;
1495   delete this.fulfill;
1496   delete this.reject;
1497 
1498   /**
1499    * A promise that resolves to the JSON representation of this WebElement's
1500    * ID, as defined by the WebDriver wire protocol.
1501    * @private {!webdriver.promise.Promise}
1502    * @see http://code.google.com/p/selenium/wiki/JsonWireProtocol
1503    */
1504   this.id_ = webdriver.promise.when(id, function(id) {
1505     if (id instanceof webdriver.WebElement) {
1506       return id.id_;
1507     } else if (goog.isDef(id[webdriver.WebElement.ELEMENT_KEY])) {
1508       return id;
1509     }
1510 
1511     var json = {};
1512     json[webdriver.WebElement.ELEMENT_KEY] = id;
1513     return json;
1514   });
1515 
1516   // This WebElement should not be resolved until its ID has been
1517   // fully resolved.
1518   this.id_.then(fulfill, reject);
1519 };
1520 goog.inherits(webdriver.WebElement, webdriver.promise.Deferred);

1965 /**
1966  * Represents a modal dialog such as {@code alert}, {@code confirm}, or
1967  * {@code prompt}. Provides functions to retrieve the message displayed with
1968  * the alert, accept or dismiss the alert, and set the response text (in the
1969  * case of {@code prompt}).
1970  * @param {!webdriver.WebDriver} driver The driver controlling the browser this
1971  *     alert is attached to.
1972  * @param {!(string|webdriver.promise.Promise)} text Either the message text
1973  *     displayed with this alert, or a promise that will be resolved to said
1974  *     text.
1975  * @constructor
1976  * @extends {webdriver.promise.Deferred}
1977  */
1978 webdriver.Alert = function(driver, text) {
1979   goog.base(this, null, driver.controlFlow());
1980 
1981   /** @private {!webdriver.WebDriver} */
1982   this.driver_ = driver;
1983 
1984   // This class is responsible for resolving itself; delete the resolve and
1985   // reject methods so they may not be accessed by consumers of this class.
1986   var fulfill = goog.partial(this.fulfill, this);
1987   var reject = this.reject;
1988   delete this.promise;
1989   delete this.fulfill;
1990   delete this.reject;
1991 
1992   /** @private {!webdriver.promise.Promise} */
1993   this.text_ = webdriver.promise.when(text);
1994 
1995   // Make sure this instance is resolved when its displayed text is.
1996   this.text_.then(fulfill, reject);
1997 };
1998 goog.inherits(webdriver.Alert, webdriver.promise.Deferred);
