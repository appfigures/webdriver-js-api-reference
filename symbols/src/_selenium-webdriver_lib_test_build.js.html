 16 'use strict';
 17 
 18 var spawn = require('child_process').spawn,
 19     fs = require('fs'),
 20     path = require('path');
 21 
 22 var promise = require('../..').promise,
 23     base = require('../../_base');
 24 
 25 var projectRoot = path.normalize(path.join(__dirname, '../../../../..'));
 26 
 27 
 28 function checkIsDevMode() {
 29   if (!base.isDevMode()) {
 30     throw Error('Cannot execute build; not running in dev mode');
 31   }
 32 }
 33 
 34 
 35 /**
 36  * Targets that have been previously built.
 37  * @type {!Object}
 38  */
 39 var builtTargets = {};
 40 
 41 
 42 /**
 43  * @param {!Array.<string>} targets The targets to build.
 44  * @throws {Error} If not running in dev mode.
 45  * @constructor
 46  */
 47 var Build = function(targets) {
 48   checkIsDevMode();
 49   this.targets_ = targets;
 50 };
 51 
 52 
 53 /** @private {boolean} */
 54 Build.prototype.cacheResults_ = false;
 55 
 56 
 57 /**
 58  * Configures this build to only execute if it has not previously been
 59  * run during the life of the current process.
 60  * @return {!Build} A self reference.
 61  */
 62 Build.prototype.onlyOnce = function() {
 63   this.cacheResults_ = true;
 64   return this;
 65 };
 66 
 67 
 68 /**
 69  * Executes the build.
 70  * @return {!webdriver.promise.Promise} A promise that will be resolved when
 71  *     the build has completed.
 72  * @throws {Error} If no targets were specified.
 73  */
 74 Build.prototype.go = function() {
 75   var targets = this.targets_;
 76   if (!targets.length) {
 77     throw Error('No targets specified');
 78   }
 79 
 80   // Filter out cached results.
 81   if (this.cacheResults_) {
 82     targets = targets.filter(function(target) {
 83       return !builtTargets.hasOwnProperty(target);
 84     });
 85 
 86     if (!targets.length) {
 87       return promise.fulfilled();
 88     }
 89   }
 90 
 91   console.log('\nBuilding', targets.join(' '), '...');
 92 
 93   var cmd, args = targets;
 94   if (process.platform === 'win32') {
 95     cmd = 'cmd.exe';
 96     args.unshift('/c', path.join(projectRoot, 'go.bat'));
 97   } else {
 98     cmd = path.join(projectRoot, 'go');
 99   }
100 
101   var result = promise.defer();
102   spawn(cmd, args, {
103     cwd: projectRoot,
104     env: process.env,
105     stdio: ['ignore', process.stdout, process.stderr]
106   }).on('exit', function(code, signal) {
107     if (code === 0) {
108       targets.forEach(function(target) {
109         builtTargets[target] = 1;
110       });
111       return result.fulfill();
112     }
113 
114     var msg = 'Unable to build artifacts';
115     if (code) {  // May be null.
116       msg += '; code=' + code;
117     }
118     if (signal) {
119       msg += '; signal=' + signal;
120     }
121 
122     result.reject(Error(msg));
123   });
124 
125   return result.promise;
126 };
127 
128 
129 // PUBLIC API
130 
131 
132 /**
133  * Creates a build of the listed targets.
134  * @param {...string} var_args The targets to build.
135  * @return {!Build} The new build.
136  * @throws {Error} If not running in dev mode.
137  */
138 exports.of = function(var_args) {
139   var targets = Array.prototype.slice.call(arguments, 0);
140   return new Build(targets);
141 };
142 
143 
144 /**
145  * @return {string} Absolute path of the project's root directory.
146  * @throws {Error} If not running in dev mode.
147  */
148 exports.projectRoot = function() {
149   checkIsDevMode();
150   return projectRoot;
151 };
