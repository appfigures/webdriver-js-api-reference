 16 'use strict';
 17 
 18 var fs = require('fs'),
 19     path = require('path');
 20 
 21 var resourceRoot = require('../../_base').isDevMode() ?
 22     require('./build').projectRoot() :
 23     path.join(__dirname, 'data');
 24 
 25 
 26 // PUBLIC API
 27 
 28 
 29 /**
 30  * Locates a test resource.
 31  * @param {string} resourcePath Path of the resource to locate.
 32  * @param {string} filePath The file to locate from the root of the project.
 33  * @return {string} The full path for the file, if it exists.
 34  * @throws {Error} If the file does not exist.
 35  */
 36 exports.locate = function(filePath) {
 37   var fullPath = path.normalize(path.join(resourceRoot, filePath));
 38   if (!fs.existsSync(fullPath)) {
 39     throw Error('File does not exist: ' + filePath);
 40   }
 41   return fullPath;
 42 };
