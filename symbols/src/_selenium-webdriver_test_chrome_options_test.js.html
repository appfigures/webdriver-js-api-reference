 16 'use strict';
 17 
 18 var fs = require('fs');
 19 
 20 var webdriver = require('../..'),
 21     chrome = require('../../chrome'),
 22     proxy = require('../../proxy'),
 23     assert = require('../../testing/assert');
 24 
 25 var test = require('../../lib/test');
 26 
 27 
 28 describe('chrome.Options', function() {
 29 
 30   describe('fromCapabilities', function() {
 31 
 32     it('should return a new Options instance if none were defined',
 33        function() {
 34          var options = chrome.Options.fromCapabilities(
 35              new webdriver.Capabilities());
 36          assert(options).instanceOf(chrome.Options);
 37        });
 38 
 39     it('should return options instance if present', function() {
 40       var options = new chrome.Options();
 41       var caps = options.toCapabilities();
 42       assert(caps).instanceOf(webdriver.Capabilities);
 43       assert(chrome.Options.fromCapabilities(caps)).equalTo(options);
 44     });
 45 
 46     it('should rebuild options from wire representation', function() {
 47       var caps = webdriver.Capabilities.chrome().set('chromeOptions', {
 48         args: ['a', 'b'],
 49         extensions: [1, 2],
 50         binary: 'binaryPath',
 51         logFile: 'logFilePath',
 52         detach: true,
 53         localState: 'localStateValue',
 54         prefs: 'prefsValue'
 55       });
 56 
 57       var options = chrome.Options.fromCapabilities(caps);
 58 
 59       assert(options.args_.length).equalTo(2);
 60       assert(options.args_[0]).equalTo('a');
 61       assert(options.args_[1]).equalTo('b');
 62       assert(options.extensions_.length).equalTo(2);
 63       assert(options.extensions_[0]).equalTo(1);
 64       assert(options.extensions_[1]).equalTo(2);
 65       assert(options.binary_).equalTo('binaryPath');
 66       assert(options.logFile_).equalTo('logFilePath');
 67       assert(options.detach_).equalTo(true);
 68       assert(options.localState_).equalTo('localStateValue');
 69       assert(options.prefs_).equalTo('prefsValue');
 70     });
 71 
 72     it('should rebuild options from incomplete wire representation',
 73         function() {
 74           var caps = webdriver.Capabilities.chrome().set('chromeOptions', {
 75             logFile: 'logFilePath'
 76           });
 77 
 78           var options = chrome.Options.fromCapabilities(caps);
 79           var json = options.toJSON();
 80 
 81           assert(json.args.length).equalTo(0);
 82           assert(json.binary).isUndefined();
 83           assert(json.detach).isFalse();
 84           assert(json.extensions.length).equalTo(0);
 85           assert(json.localState).isUndefined();
 86           assert(json.logFile).equalTo('logFilePath');
 87           assert(json.prefs).isUndefined();
 88         });
 89 
 90     it('should extract supported WebDriver capabilities', function() {
 91       var proxyPrefs = proxy.direct();
 92       var logPrefs = {};
 93       var caps = webdriver.Capabilities.chrome().
 94           set(webdriver.Capability.PROXY, proxyPrefs).
 95           set(webdriver.Capability.LOGGING_PREFS, logPrefs);
 96 
 97       var options = chrome.Options.fromCapabilities(caps);
 98       assert(options.proxy_).equalTo(proxyPrefs);
 99       assert(options.logPrefs_).equalTo(logPrefs);
100     });
101   });
102 
103   describe('addArguments', function() {
104     it('takes var_args', function() {
105       var options = new chrome.Options();
106       assert(options.args_.length).equalTo(0);
107 
108       options.addArguments('a', 'b');
109       assert(options.args_.length).equalTo(2);
110       assert(options.args_[0]).equalTo('a');
111       assert(options.args_[1]).equalTo('b');
112     });
113 
114     it('flattens input arrays', function() {
115       var options = new chrome.Options();
116       assert(options.args_.length).equalTo(0);
117 
118       options.addArguments(['a', 'b'], 'c', [1, 2], 3);
119       assert(options.args_.length).equalTo(6);
120       assert(options.args_[0]).equalTo('a');
121       assert(options.args_[1]).equalTo('b');
122       assert(options.args_[2]).equalTo('c');
123       assert(options.args_[3]).equalTo(1);
124       assert(options.args_[4]).equalTo(2);
125       assert(options.args_[5]).equalTo(3);
126     });
127   });
128 
129   describe('addExtensions', function() {
130     it('takes var_args', function() {
131       var options = new chrome.Options();
132       assert(options.extensions_.length).equalTo(0);
133 
134       options.addExtensions('a', 'b');
135       assert(options.extensions_.length).equalTo(2);
136       assert(options.extensions_[0]).equalTo('a');
137       assert(options.extensions_[1]).equalTo('b');
138     });
139 
140     it('flattens input arrays', function() {
141       var options = new chrome.Options();
142       assert(options.extensions_.length).equalTo(0);
143 
144       options.addExtensions(['a', 'b'], 'c', [1, 2], 3);
145       assert(options.extensions_.length).equalTo(6);
146       assert(options.extensions_[0]).equalTo('a');
147       assert(options.extensions_[1]).equalTo('b');
148       assert(options.extensions_[2]).equalTo('c');
149       assert(options.extensions_[3]).equalTo(1);
150       assert(options.extensions_[4]).equalTo(2);
151       assert(options.extensions_[5]).equalTo(3);
152     });
153   });
154 
155   describe('toJSON', function() {
156     it('base64 encodes extensions', function() {
157       var expected = fs.readFileSync(__filename, 'base64');
158       var wire = new chrome.Options().addExtensions(__filename).toJSON();
159       assert(wire.extensions.length).equalTo(1);
160       assert(wire.extensions[0]).equalTo(expected);
161     });
162   });
163 
164   describe('toCapabilities', function() {
165     it('returns a new capabilities object if one is not provided', function() {
166       var options = new chrome.Options();
167       var caps = options.toCapabilities();
168       assert(caps.get('browserName')).equalTo('chrome');
169       assert(caps.get('chromeOptions')).equalTo(options);
170     });
171 
172     it('adds to input capabilities object', function() {
173       var caps = webdriver.Capabilities.firefox();
174       var options = new chrome.Options();
175       assert(options.toCapabilities(caps)).equalTo(caps);
176       assert(caps.get('browserName')).equalTo('firefox');
177       assert(caps.get('chromeOptions')).equalTo(options);
178     });
179 
180     it('sets generic driver capabilities', function() {
181       var proxyPrefs = {};
182       var loggingPrefs = {};
183       var options = new chrome.Options().
184           setLoggingPreferences(loggingPrefs).
185           setProxy(proxyPrefs);
186 
187       var caps = options.toCapabilities();
188       assert(caps.get('proxy')).equalTo(proxyPrefs);
189       assert(caps.get('loggingPrefs')).equalTo(loggingPrefs);
190     });
191   });
192 });

194 test.suite(function(env) {
195   env.autoCreateDriver = false;
196 
197   describe('options', function() {
198     test.it('can start Chrome with custom args', function() {
199       var options = new chrome.Options().
200           addArguments('user-agent=foo;bar');
201 
202       var driver = env.builder().
203           setChromeOptions(options).
204           build();
205 
206       driver.get(test.Pages.ajaxyPage);
207 
208       var userAgent = driver.executeScript(
209           'return window.navigator.userAgent');
210       assert(userAgent).equalTo('foo;bar');
211     });
212   });
213 }, {browsers: ['chrome']});
