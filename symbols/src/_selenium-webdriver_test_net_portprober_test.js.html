 16 'use strict';
 17 
 18 var assert = require('assert'),
 19     net = require('net');
 20 
 21 var promise = require('../..').promise,
 22     portprober = require('../../net/portprober');
 23 
 24 describe('isFree', function() {
 25 
 26   var server;
 27 
 28   beforeEach(function() {
 29     server = net.createServer(function(){});
 30   })
 31 
 32   afterEach(function(done) {
 33     if (!server) return done();
 34     server.close(function() {
 35       done();
 36     });
 37   });
 38 
 39   it('should work for INADDR_ANY', function(done) {
 40     server.listen(0, function() {
 41       var port = server.address().port;
 42       assertPortNotfree(port).then(function() {
 43         var done = promise.defer();
 44         server.close(function() {
 45           server = null;
 46           done.fulfill(assertPortIsFree(port));
 47         });
 48         return done.promise;
 49       }).then(function() { done(); }, done);
 50     });
 51   });
 52 
 53   it('should work for a specific host', function(done) {
 54     var host = '127.0.0.1';
 55     server.listen(0, host, function() {
 56       var port = server.address().port;
 57       assertPortNotfree(port, host).then(function() {
 58         var done = promise.defer();
 59         server.close(function() {
 60           server = null;
 61           done.fulfill(assertPortIsFree(port, host));
 62         });
 63         return done.promise;
 64       }).then(function() { done(); }, done);
 65     });
 66   });
 67 });
 68 
 69 describe('findFreePort', function() {
 70   var server;
 71 
 72   beforeEach(function() {
 73     server = net.createServer(function(){});
 74   })
 75 
 76   afterEach(function(done) {
 77     if (!server) return done();
 78     server.close(function() {
 79       done();
 80     });
 81   });
 82 
 83   it('should work for INADDR_ANY', function(done) {
 84     portprober.findFreePort().then(function(port) {
 85       server.listen(port, function() {
 86         assertPortNotfree(port).then(function() {
 87           var done = promise.defer();
 88           server.close(function() {
 89             server = null;
 90             done.fulfill(assertPortIsFree(port));
 91           });
 92           return done.promise;
 93         }).then(function() { done(); }, done);
 94       });
 95     });
 96   });
 97 
 98   it('should work for a specific host', function(done) {
 99     var host = '127.0.0.1';
100     portprober.findFreePort(host).then(function(port) {
101       server.listen(port, host, function() {
102         assertPortNotfree(port, host).then(function() {
103           var done = promise.defer();
104           server.close(function() {
105             server = null;
106             done.fulfill(assertPortIsFree(port, host));
107           });
108           return done.promise;
109         }).then(function() { done(); }, done);
110       });
111     });
112   });
113 });
114 
115 
116 function assertPortIsFree(port, opt_host) {
117   return portprober.isFree(port, opt_host).then(function(free) {
118     assert.ok(free, 'port should be free');
119   });
120 }
121 
122 
123 function assertPortNotfree(port, opt_host) {
124   return portprober.isFree(port, opt_host).then(function(free) {
125     assert.ok(!free, 'port should is not free');
126   });
127 }
