105   BASIC_AUTH: path.join(WEB_ROOT, 'basicAuth'),
106   MANIFEST: path.join(WEB_ROOT, 'manifest'),
107   REDIRECT: path.join(WEB_ROOT, 'redirect'),
108   PAGE: path.join(WEB_ROOT, 'page'),
109   SLEEP: path.join(WEB_ROOT, 'sleep'),
110   UPLOAD: path.join(WEB_ROOT, 'upload')

114 /**
115  * HTTP request handler.
116  * @param {!http.ServerRequest} request The request object.
117  * @param {!http.ServerResponse} response The response object.
118  */
119 function onRequest(request, response) {
120   if (request.method !== 'GET' && request.method !== 'HEAD') {
121     response.writeHead(405, {'Allowed': 'GET,HEAD'});
122     return response.end();
123   }
124 
125   var pathname = path.resolve(url.parse(request.url).pathname);
126   if (pathname === '/') {
127     return sendIndex(request, response);
128   }
129 
130   if (pathname === '/favicon.ico') {
131     response.writeHead(204);
132     return response.end();
133   }
134 
135   switch (pathname) {
136     case Path.BASIC_AUTH: return sendBasicAuth(response);
137     case Path.MANIFEST: return sendManifest(response);
138     case Path.PAGE: return sendInifinitePage(request, response);
139     case Path.REDIRECT: return redirectToResultPage(response);
140     case Path.SLEEP: return sendDelayedResponse(request, response);
141     case Path.UPLOAD: return sendUpload(response);
142   }
143 
144   if (string.startsWith(pathname, Path.PAGE + '/')) {
145     return sendInifinitePage(request, response);
146   }
147 
148   if ((string.startsWith(pathname, WEB_ROOT) ||
149        string.startsWith(pathname, JS_ROOT)) &&
150       string.endsWith(pathname, '.appcache')) {
151     return sendManifest(response);
152   }
153 
154   if (string.startsWith(pathname, WEB_ROOT)) {
155     if (!isDevMode) {
156       pathname = pathname.substring(WEB_ROOT.length);
157     }
158   } else if (string.startsWith(pathname, JS_ROOT)) {
159     if (!isDevMode) {
160       sendSimpleError(response, 404, request.url);
161       return;
162     }
163     pathname = pathname.substring(JS_ROOT);
164   }
165 
166   try {
167     var fullPath = resources.locate(pathname);
168   } catch (ex) {
169     fullPath = '';
170   }
171 
172   if (fullPath.lastIndexOf(baseDirectory, 0) == -1) {
173     sendSimpleError(response, 404, request.url);
174     return;
175   }
176 
177   fs.stat(fullPath, function(err, stats) {
178     if (err) {
179       sendIOError(request, response, err);
180     } else if (stats.isDirectory()) {
181       sendDirectoryListing(request, response, fullPath);
182     } else if (stats.isFile()) {
183       sendFile(request, response, fullPath);
184     } else {
185       sendSimpleError(response, 404, request.url);
186     }
187   });
188 }

199 function sendInifinitePage(request, response) {
200   setTimeout(function() {
201     var pathname = url.parse(request.url).pathname;
202     var lastIndex = pathname.lastIndexOf('/');
203     var pageNumber =
204         (lastIndex == -1 ? 'Unknown' : pathname.substring(lastIndex + 1));
205     var body = [
206       '<!DOCTYPE html>',
207       '<title>Page', pageNumber, '</title>',
208       'Page number <span id="pageNumber">', pageNumber, '</span>',
209       '<p><a href="../xhtmlTest.html" target="_top">top</a>'
210     ].join('');
211     response.writeHead(200, {
212       'Content-Length': Buffer.byteLength(body, 'utf8'),
213       'Content-Type': 'text/html; charset=utf-8'
214     });
215     response.end(body);
216   }, 500);
217 }

220 function sendDelayedResponse(request, response) {
221   var duration = 0;
222   var query = url.parse(request.url).query || '';
223   var match = query.match(/\btime=(\d+)/);
224   if (match) {
225     duration = parseInt(match[1]);
226   }
227 
228   setTimeout(function() {
229     var body = [
230       '<!DOCTYPE html>',
231       '<title>Done</title>',
232       '<body>Slept for ', duration, 's</body>'
233     ].join('');
234     response.writeHead(200, {
235       'Content-Length': Buffer.byteLength(body, 'utf8'),
236       'Content-Type': 'text/html; charset=utf-8',
237       'Cache-Control': 'no-cache',
238       'Pragma': 'no-cache',
239       'Expires': 0
240     });
241     response.end(body);
242   }, duration * 1000);
243 }

246 /**
247  * Sends an error in response to an I/O operation.
248  * @param {!http.ServerRequest} request The request object.
249  * @param {!http.ServerResponse} response The response object.
250  * @param {!Error} err The I/O error.
251  */
252 function sendIOError(request, response, err) {
253   var code = 500;
254   if (err.code === 'ENOENT') {
255     code = 404;
256   } else if (err.code === 'EACCES') {
257     code = 403;
258   }
259   sendSimpleError(response, code, request.url);
260 }

263 /**
264  * Sends a simple error message to the client and instructs it to close the
265  * connection.
266  * @param {!http.ServerResponse} response The response to populate.
267  * @param {number} code The numeric HTTP code to send.
268  * @param {string} message The error message.
269  */
270 function sendSimpleError(response, code, message) {
271   response.writeHead(code, {
272     'Content-Type': 'text/html; charset=utf-8',
273     'Connection': 'close'
274   });
275   response.end(
276       '<!DOCTYPE html><h1>' + code + ' ' + http.STATUS_CODES[code] +
277       '</h1><hr/>' + message);
278 }

281 var MimeType = {
282   'css': 'text/css',
283   'gif': 'image/gif',
284   'html': 'text/html',
285   'js': 'application/javascript',
286   'png': 'image/png',
287   'svg': 'image/svg+xml',
288   'txt': 'text/plain',
289   'xhtml': 'application/xhtml+xml',

295 /**
296  * Responds to a request for an individual file.
297  * @param {!http.ServerRequest} request The request object.
298  * @param {!http.ServerResponse} response The response object.
299  * @param {string} filePath Path to the file to return.
300  */
301 function sendFile(request, response, filePath) {
302   fs.readFile(filePath, function(err, buffer) {
303     if (err) {
304       sendIOError(request, response, err);
305       return;
306     }
307     var index = filePath.lastIndexOf('.');
308     var type = MimeType[index < 0 ? '' : filePath.substring(index + 1)];
309     var headers = {'Content-Length': buffer.length};
310     if (type) headers['Content-Type'] = type;
311     response.writeHead(200, headers);
312     response.end(buffer);
313   });
314 }

317 /**
318  * Responds to a request for the file server's main index.
319  * @param {!http.ServerRequest} request The request object.
320  * @param {!http.ServerResponse} response The response object.
321  */
322 function sendIndex(request, response) {
323   var pathname = url.parse(request.url).pathname;
324 
325   var host = request.headers.host;
326   if (!host) {
327     host = server.host();
328   }
329 
330   var requestUrl = ['http://' + host + pathname].join('');
331 
332   function createListEntry(path) {
333     var url = requestUrl + path;
334     return ['<li><a href="', url, '">', path, '</a>'].join('');
335   }
336 
337   var data = ['<!DOCTYPE html><h1>/</h1><hr/><ul>',
338               createListEntry('common')];
339   if (isDevMode) {
340     data.push(createListEntry('javascript'));
341   }
342   data.push('</ul>');
343   data = data.join('');
344 
345   response.writeHead(200, {
346     'Content-Type': 'text/html; charset=UTF-8',
347     'Content-Length': Buffer.byteLength(data, 'utf8')
348   });
349   response.end(data);
350 }

353 /**
354  * Responds to a request for a directory listing.
355  * @param {!http.ServerRequest} request The request object.
356  * @param {!http.ServerResponse} response The response object.
357  * @param {string} dirPath Path to the directory to generate a listing for.
358  */
359 function sendDirectoryListing(request, response, dirPath) {
360   var pathname = url.parse(request.url).pathname;
361 
362   var host = request.headers.host;
363   if (!host) {
364     host = server.host();
365   }
366 
367   var requestUrl = ['http://' + host + pathname].join('');
368   if (requestUrl[requestUrl.length - 1] !== '/') {
369     response.writeHead(303, {'Location': requestUrl + '/'});
370     return response.end();
371   }
372 
373   fs.readdir(dirPath, function(err, files) {
374     if (err) {
375       sendIOError(request, response, err);
376       return;
377     }
378 
379     var data = ['<!DOCTYPE html><h1>', pathname, '</h1><hr/><ul>'];
380     if (pathname !== '/') {
381       data.push(createListEntry('../'));
382     }
383     processNextFile();
384 
385     function processNextFile() {
386       var file = files.shift();
387       if (file) {
388         fs.stat(path.join(dirPath, file), function(err, stats) {
389           if (err) {
390             sendIOError(request, response, err);
391             return;
392           }
393 
394           data.push(createListEntry(
395               stats.isDirectory() ? file + '/' : file));
396           processNextFile();
397         });
398       } else {
399         data = new Buffer(data.join(''), 'utf-8');
400         response.writeHead(200, {
401           'Content-Type': 'text/html; charset=utf-8',
402           'Content-Length': data.length
403         });
404         response.end(data);
405       }
406     }
407 
408     function createListEntry(path) {
409       var url = requestUrl + path;
410       return ['<li><a href="', url, '">', path, '</a>'].join('');
411     }
412   });
413 }
