 27 var Browser = {
 28   ANDROID: 'android',
 29   CHROME: 'chrome',
 30   IE: 'internet explorer',
 31   // Shorthand for IPAD && IPHONE when using the browsers predciate.
 32   IOS: 'iOS',
 33   IPAD: 'iPad',
 34   IPHONE: 'iPhone',
 35   FIREFOX: 'firefox',
 36   OPERA: 'opera',
 37   PHANTOMJS: 'phantomjs',
 38   SAFARI: 'safari',

 40   // Browsers that should always be tested via the java Selenium server.
 41   REMOTE_CHROME: 'remote.chrome',
 42   REMOTE_PHANTOMJS: 'remote.phantomjs'

 46 /**
 47  * Browsers with native support.
 48  * @type {!Array.<string>}
 49  */
 50 var NATIVE_BROWSERS = [
 51   Browser.CHROME,
 52   Browser.PHANTOMJS
 53 ];

 56 var browsersToTest = (function() {
 57   var browsers = process.env['SELENIUM_BROWSER'] || Browser.CHROME;
 58   browsers = browsers.split(',');
 59   browsers.forEach(function(browser) {
 60     if (browser === Browser.IOS) {
 61       throw Error('Invalid browser name: ' + browser);
 62     }
 63 
 64     for (var name in Browser) {
 65       if (Browser.hasOwnProperty(name) && Browser[name] === browser) {
 66         return;
 67       }
 68     }
 69 
 70     throw Error('Unrecognized browser: ' + browser);
 71   });
 72   return browsers;
 73 })();

 76 /**
 77  * Creates a predicate function that ignores tests for specific browsers.
 78  * @param {string} currentBrowser The name of the current browser.
 79  * @param {!Array.<!Browser>} browsersToIgnore The browsers to ignore.
 80  * @return {function(): boolean} The predicate function.
 81  */
 82 function browsers(currentBrowser, browsersToIgnore) {
 83   return function() {
 84     var checkIos =
 85         currentBrowser === Browser.IPAD || currentBrowser === Browser.IPHONE;
 86     return browsersToIgnore.indexOf(currentBrowser) != -1 ||
 87         (checkIos && browsersToIgnore.indexOf(Browser.IOS) != -1);
 88   };
 89 }

 92 /**
 93  * @param {string} browserName The name to use.
 94  * @param {remote.DriverService} server The server to use, if any.
 95  * @constructor
 96  */
 97 function TestEnvironment(browserName, server) {
 98   var name = browserName;
 99   if (name.lastIndexOf('remote.', 0) == 0) {
100     name = name.substring('remote.'.length);
101   }
102 
103   var autoCreate = true;
104   this.__defineGetter__(
105       'autoCreateDriver', function() { return autoCreate; });
106   this.__defineSetter__(
107       'autoCreateDriver', function(auto) { autoCreate = auto; });
108 
109   this.__defineGetter__('browser', function() { return name; });
110 
111   var driver;
112   this.__defineGetter__('driver', function() { return driver; });
113 
114   this.browsers = function(var_args) {
115     var browsersToIgnore = Array.prototype.slice.apply(arguments, [0]);
116     var remoteVariants = [];
117     browsersToIgnore.forEach(function(browser) {
118       if (browser.lastIndexOf('remote.', 0) === 0) {
119         remoteVariants.push(browser.substring('remote.'.length));
120       }
121     });
122     browsersToIgnore = browsersToIgnore.concat(remoteVariants);
123     return browsers(browserName, browsersToIgnore);
124   };
125 
126   this.builder = function() {
127     assert.ok(!driver, 'Can only have one driver at a time');
128     var builder = new webdriver.Builder();
129     var realBuild = builder.build;
130 
131     builder.build = function() {
132       builder.getCapabilities().
133           set(webdriver.Capability.BROWSER_NAME, name);
134 
135       if (server) {
136         builder.usingServer(server.address());
137       }
138       return driver = realBuild.call(builder);
139     };
140 
141     return builder;
142   };
143 
144   this.createDriver = function() {
145     if (!driver) {
146       driver = this.builder().build();
147     }
148     return driver;
149   };
150 
151   this.refreshDriver = function() {
152     if (driver) {
153       driver.quit();
154       driver = null;
155     }
156     this.createDriver();
157   };
158 
159   this.dispose = function() {
160     if (driver) {
161       driver.quit();
162       driver = null;
163     }
164   };
165 
166   this.waitForTitleToBe = function(expected) {
167     driver.wait(function() {
168       return driver.getTitle().then(function(title) {
169         return title === expected;
170       });
171     }, 5000, 'Waiting for title to be ' + expected);
172   };
173 }

176 var seleniumServer;
177 var inSuite = false;
178 
179 
180 /**
181  * Expands a function to cover each of the target browsers.
182  * @param {function(!TestEnvironment)} fn The top level suite
183  *     function.
184  * @param {{browsers: !Array.<string>}=} opt_options Suite specific options.
185  */
186 function suite(fn, opt_options) {
187   assert.ok(!inSuite, 'You may not nest suite calls');
188   inSuite = true;
189 
190   var suiteOptions = opt_options || {};
191   var browsers = suiteOptions.browsers;
192   if (browsers) {
193     // Filter out browser specific tests when that browser is not currently
194     // selected for testing.
195     browsers = browsers.filter(function(browser) {
196       if (browsersToTest.indexOf(browser) != -1) {
197         return true;
198       }
199       return browsersToTest.indexOf(
200           browser.substring('remote.'.length)) != -1;
201     });
202   } else {
203     browsers = browsersToTest;
204   }
205 
206   try {
207     browsers.forEach(function(browser) {
208 
209       testing.describe('[' + browser + ']', function() {
210         var serverToUse = null;
211 
212         if (NATIVE_BROWSERS.indexOf(browser) == -1) {
213           serverToUse = seleniumServer;
214           if (!serverToUse) {
215             serverToUse = seleniumServer = new seleniumserver.Server();
216           }
217           testing.before(seleniumServer.start.bind(seleniumServer, 60 * 1000));
218         }
219 
220         var env = new TestEnvironment(browser, serverToUse);
221 
222         testing.beforeEach(function() {
223           if (env.autoCreateDriver) {
224             env.createDriver();
225           }
226         });
227 
228         testing.after(function() {
229           env.dispose();
230         });
231 
232         fn(env);
233       });
234     });
235   } finally {
236     inSuite = false;
237   }
238 }
